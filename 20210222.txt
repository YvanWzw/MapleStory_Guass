//要取得[a, b)的随机整数，使用(rand() % (b - a)) + a;
//要取得[a, b]的随机整数，使用(rand() % (b - a + 1)) + a;
//要取得(a, b]的随机整数，使用(rand() % (b - a)) + a + 1;
//通用公式:a + rand() % n；其中的a是起始值，n是整数的范围。
//要取得a到b之间的随机整数，另一种表示：a + (int)b * rand() / (RAND_MAX + 1)。
//要取得0～1之间的浮点数，可以使用rand() / double(RAND_MAX)。
#include<iostream>
#include <cstdlib>
#include <ctime>
using namespace std;
int main()
{
	unsigned seed;
	seed = time(0);
	srand(seed);

	int a[19];
	for (int i = 0; i < 19; i++)
	{
		a[i] = i + 1;
	}
	int sed1 = rand() % 19 + 1;
	int sed = rand() % 2;
	int sed2;
	if (sed == 0) {
		sed2 = rand() % sed1 + 1;
	}
	else if (sed == 1 && sed1 != 19)
	{
		sed2 = rand() % (19 - sed1) + sed1 + 1;
	}
	else if (sed == 1 && sed1 == 19)
	{
		sed2 = rand() % 18 + 1;
	}

	return 0;
}
